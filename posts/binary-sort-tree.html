# 删除结点 p

// f 为 p 的父结点
// s 为中序序列中 p 的前驱结点
// q 为 s 的父结点

// one way
f->lchild = p->lchild;
s->rchild = p->rchild;
free(p);

// another way
p->data = s->data;
q->rchild = s->lchild;
free(s);
